9   import io.jsonwebtoken.Claims;
10  import io.jsonwebtoken.Jwts;
11  import io.jsonwebtoken.SignatureAlgorithm;
12  
13  @Component
14  public class JwtTokenProvider {
15  	public String generateToken(Authentication authentication) {
16  		String username = authentication.getName();
17  		Date currentDate = new Date(System.currentTimeMillis());
18  		Date expireDate = new Date(currentDate.getTime() + JwtConstants.JWT_EXPIRATION);
19  		String token = Jwts.builder()
20  				.setSubject(username)
21  				.setIssuedAt(currentDate)
22  				.setExpiration(expireDate)
23  				.signWith(SignatureAlgorithm.HS512, JwtConstants.JWT_SECRET)
24  				.compact();
25  		return token;
26  	}
27  
28  	//extract JWT to get username value
29  	public String getUsernameFromJwt(String token) {
30  		Claims claims = Jwts
31  				.parser()
32  				.setSigningKey(JwtConstants.JWT_SECRET)
33  				.parseClaimsJws(token)
34  				.getBody();
35  		return claims.getSubject();
36  	}
