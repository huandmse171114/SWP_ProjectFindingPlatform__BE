117 	@Override
118 	public ProjectResponseModel getById(int id) {
119 		Project project = get(id);
120 
121 		if (project == null) return null;
122 
123 		List<SkillModel> skills = new ArrayList<>();
124 		skillService.getNameAndLevelByProjectId(id)
125 			.forEach(skill -> skills.add(
126 					SkillModel
127 						.builder()
128 							.name(skill.getName())
129 							.level(skill.getLevel())
130 						.build()
131 				)
132 			);
133 
134 		String status = ProjectStatus.nameOf(project.getStatus());
135 		Date dueDate = Utils.addDate(project.getPublishDate(), project.getDeliverDays());
136 
137 		List<CategoryDTO> categoriesObj = categoryService.getNameByProjectId(id);
138 		List<String> categories = new ArrayList<>();
139 		categoriesObj.forEach(each -> categories.add(each.getName()));
140 
141 		return ProjectResponseModel
142 				.builder()
143 					.id(id)
144 					.name(project.getName())
145 					.publishDate(project.getPublishDate().toString())
146 					.deliverDays(project.getDeliverDays())
147 					.publisher(
148 						publisherService.get(
149 							project.getPublisherId()
150 						)
151 					)
152 					.wage(project.getWage())
153 					.dueDate(dueDate.toString())
154 					.status(status)
155 					// .skills(skills)
156 					// .categories(categories)
157 					.description(project.getDescription())
158 				.build();
159 	}

202 	@Override
203 	public ProjectResponseModel getModel(int id) {
204 		Project p = get(id);
205 
206 		if (p == null) return null;
207 
208 		List<ProjectSkillResponseModel> skills = new ArrayList<>();
209 		(p.getSkills()).forEach(
210 			skill -> skills.add(
211 				ProjectSkillResponseModel
212 					.builder()
213 						.id(skill.getSkill().getId())
214 						.projectSkillId(skill.getId())
215 						.name(skill.getSkill().getName())
216 						.level(skill.getLevel())
217 						.status(
218 							ProjectSkillStatus.nameOf(
219 								skill.getStatus()
220 							)
221 						)
222 					.build()
223 			)
224 		);
225 
226 		List<ProjectCategoryResponseModel> categories = new ArrayList<>();
227 		List<String> categoriesStr = new ArrayList<>();
228 		(p.getCategories()).forEach(
229 			category -> {
230 				String name = category.getCategory().getName();
231 				categories.add(
232 					ProjectCategoryResponseModel
233 						.builder()
234 							.id(category.getCategory().getId())
235 							.projectCategoryId(category.getId())
236 							.name(name)
237 							.status(
238 								ProjectCategoryStatus.nameOf(
239 									category.getStatus()
240 								)
241 							)
242 						.build());
243 				categoriesStr.add(name);
244 			}
245 		);
246 
247 		List<ProjectDeliverableResponseModel> deliverables = new ArrayList<>();
248 		(p.getDeliverables()).forEach(
249 			deliverable -> {
250 				deliverables.add(
251 					ProjectDeliverableResponseModel
252 						.builder()
253 							.id(deliverable.getDeliverableType().getId())
254 							.projectDeliverableId(deliverable.getId())
255 							.name(deliverable.getDeliverableType().getName())
256 							.value(deliverable.getValue())
257 							.description(deliverable.getDescription())
258 							.status(
259 								ProjectDeliverableStatus.nameOf(
260 									deliverable.getStatus()
261 								)
262 							)
263 						.build()
264 				);
265 			}
266 		);
267 
268 		return ProjectResponseModel
269 				.builder()
270 					.id(p.getId())
271 					.name(p.getName())
272 					.publisher(
273 						publisherService.get(
274 							p.getPublisherId()
275 						)
276 					)
277 					.description(p.getDescription())
278 					.wage(p.getWage())
279 					.imageURL(p.getImageURL())
280 					.deliverDays(p.getDeliverDays())
281 					.publishDate(p.getPublishDate().toString())
282 					.dueDate(p.getDueDate().toString())
283 					.status(
284 						ProjectStatus.nameOf(
285 							p.getStatus()
286 						)
287 					)
288 					.skills(skills)
289 					.categories(categories)
290 					.deliverables(deliverables)
291 				.build();
292 
293 	}
