38  
39  		int projectId = model.getProjectId();
40  		Project project = projectService.get(projectId);
41  		if (project == null)
42  			return new ResponseEntity<String>(
43  				"Failed to create Payment:"
44  						+ " Project[id=" + projectId + "] not existed",
45  				HttpStatus.BAD_REQUEST
46  			);
47  
48  		int teamId = model.getTeamId();
49  		Team team = teamService.get(teamId);
50  		if (team == null)
51  			return new ResponseEntity<String>(
52  				"Failed to create Payment:"
53  						+ " Team[id=" + teamId + "] not existed",
54  				HttpStatus.BAD_REQUEST
55  			);
56  
57  		// if (!projectService.existsByTeamIdAndProjectId(teamId, projectId))

60  		// + "Project[id=" + projectId + "] does not have Team[id=" + teamId + "]",
61  		// HttpStatus.BAD_REQUEST);
62  
63  		Payment payment = Payment
64  			.builder()
65  				.projectId(projectId)
66  				.teamId(teamId)
67  				.amount(model.getAmount())
68  				.description(model.getDescription())
69  			.build();
70  
71  		service.save(payment);
